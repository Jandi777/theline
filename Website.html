  <script>

    // ---------- DATA MODEL ----------

    let appData = {

      vehicles: [

        { id: "v1", name: "Vehicle A · Ford F-150" },

        { id: "v2", name: "Vehicle B · Transit Van" }

      ],

      maintenanceLogs: [],   // { id, vehicleId, date, description, cost }

      appointments: [],      // { id, vehicleId, datetime, title, notes }

      tasks: []              // { id, text, completed }

    };

    // Helper: generate unique id

    function genId() { return Date.now() + '-' + Math.random().toString(36).substr(2, 6); }

    // Load from localStorage

    function loadFromLocalStorage() {

      const saved = localStorage.getItem("theLine_team_hub");

      if (saved) {

        try {

          const parsed = JSON.parse(saved);

          if (parsed.maintenanceLogs && parsed.appointments && parsed.tasks && parsed.vehicles) {

            appData = parsed;

          }

        } catch(e) { console.warn(e); }

      } else {

        // demo seed data

        appData.maintenanceLogs = [

          { id: genId(), vehicleId: "v1", date: "2025-03-10", description: "Oil change & filter", cost: 89.99 },

          { id: genId(), vehicleId: "v2", date: "2025-03-15", description: "Brake pad replacement", cost: 245.50 },

          { id: genId(), vehicleId: "v1", date: "2025-02-20", description: "Tire rotation", cost: 40.00 },

        ];

        appData.appointments = [

          { id: genId(), vehicleId: "v1", datetime: "2025-04-20T10:30", title: "State inspection", notes: "At certified shop" },

          { id: genId(), vehicleId: "v2", datetime: "2025-04-22T14:00", title: "Transmission fluid flush", notes: "Call ahead" },

        ];

        appData.tasks = [

          { id: genId(), text: "Order spare oil filters", completed: false },

          { id: genId(), text: "Update maintenance logs for both vehicles", completed: true },

          { id: genId(), text: "Schedule alignment for Vehicle B", completed: false },

        ];

      }

    }

    function persistToLocalStorage() {

      localStorage.setItem("theLine_team_hub", JSON.stringify(appData));

    }

    // ----- CRUD operations -----

    function addMaintenance(vehicleId, date, description, cost) {

      const newLog = { id: genId(), vehicleId, date, description, cost: cost ? parseFloat(cost) : 0 };

      appData.maintenanceLogs.push(newLog);

      persistToLocalStorage();

      renderAll();

    }

    function deleteMaintenance(logId) {

      appData.maintenanceLogs = appData.maintenanceLogs.filter(l => l.id !== logId);

      persistToLocalStorage();

      renderAll();

    }

    function addAppointment(vehicleId, datetime, title, notes) {

      const newApp = { id: genId(), vehicleId, datetime, title, notes: notes || "" };

      appData.appointments.push(newApp);

      persistToLocalStorage();

      renderAll();

    }

    function deleteAppointment(appId) {

      appData.appointments = appData.appointments.filter(a => a.id !== appId);

      persistToLocalStorage();

      renderAll();

    }

    function addTask(text) {

      if (!text.trim()) return;

      const newTask = { id: genId(), text: text.trim(), completed: false };

      appData.tasks.push(newTask);

      persistToLocalStorage();

      renderAll();

    }

    function toggleTask(taskId) {

      const task = appData.tasks.find(t => t.id === taskId);

      if (task) { task.completed = !task.completed; persistToLocalStorage(); renderAll(); }

    }

    function deleteTask(taskId) {

      appData.tasks = appData.tasks.filter(t => t.id !== taskId);

      persistToLocalStorage();

      renderAll();

    }

    // ----- RENDER ENGINE -----

    function renderMaintenanceForVehicle(vehicleId, containerId) {

      const logs = appData.maintenanceLogs.filter(log => log.vehicleId === vehicleId).sort((a,b) => new Date(b.date) - new Date(a.date));

      const container = document.getElementById(containerId);

      if (!container) return;

      if (logs.length === 0) {

        container.innerHTML = `<div class="text-gray-400 text-xs italic text-center py-2">No maintenance records yet</div>`;

        return;

      }

      container.innerHTML = logs.map(log => `

        <div class="bg-white rounded-lg p-2 border border-gray-200 flex justify-between items-start gap-2 shadow-sm">

          <div class="flex-1">

            <div class="font-medium text-gray-800 text-xs">${log.date} · ${log.description}</div>

            <div class="text-gray-500 text-xs">💰 $${log.cost ? log.cost.toFixed(2) : '0.00'}</div>

          </div>

          <button data-id="${log.id}" class="deleteMaintenanceBtn text-gray-400 hover:text-red-500 text-sm"><i class="fas fa-trash-alt"></i></button>

        </div>

      `).join('');

      // attach delete events

      container.querySelectorAll('.deleteMaintenanceBtn').forEach(btn => {

        btn.addEventListener('click', (e) => {

          e.stopPropagation();

          const id = btn.getAttribute('data-id');

          if (confirm("Delete this maintenance record?")) deleteMaintenance(id);

        });

      });

    }

    function renderAppointments() {

      const container = document.getElementById('appointmentsListContainer');

      const emptyMsg = document.getElementById('emptyAppointmentsMsg');

      if (!container) return;

      const sorted = [...appData.appointments].sort((a,b) => new Date(a.datetime) - new Date(b.datetime));

      if (sorted.length === 0) {

        container.innerHTML = '';

        emptyMsg.classList.remove('hidden');

        return;

      }

      emptyMsg.classList.add('hidden');

      container.innerHTML = sorted.map(apt => {

        const vehicle = appData.vehicles.find(v => v.id === apt.vehicleId)?.name || apt.vehicleId;

        const displayDate = new Date(apt.datetime).toLocaleString(undefined, {month:'short', day:'numeric', hour:'2-digit', minute:'2-digit'});

        return `

          <div class="bg-rose-50/40 border border-rose-100 rounded-xl p-3 flex justify-between items-start gap-2 shadow-sm">

            <div class="flex-1">

              <div class="flex flex-wrap items-center gap-x-2 gap-y-1">

                <span class="font-semibold text-gray-800 text-sm">${apt.title}</span>

                <span class="text-xs bg-white px-2 py-0.5 rounded-full text-gray-600">${vehicle}</span>

              </div>

              <div class="text-xs text-gray-500 mt-1"><i class="far fa-clock mr-1"></i> ${displayDate}</div>

              ${apt.notes ? `<div class="text-xs text-gray-400 mt-1">📌 ${apt.notes}</div>` : ''}

            </div>

            <button data-id="${apt.id}" class="deleteAppointmentBtn text-gray-400 hover:text-red-500 text-sm"><i class="fas fa-trash-alt"></i></button>

          </div>

        `;

      }).join('');

      container.querySelectorAll('.deleteAppointmentBtn').forEach(btn => {

        btn.addEventListener('click', (e) => {

          const id = btn.getAttribute('data-id');

          if (confirm("Remove this appointment?")) deleteAppointment(id);

        });

      });

      // update upcoming banner

      const upcoming = sorted.filter(apt => new Date(apt.datetime) > new Date()).slice(0, 3);

      const bannerSpan = document.getElementById('upcomingAppointmentsList');

      const bannerDiv = document.getElementById('upcomingAlert');

      if (upcoming.length > 0 && bannerSpan) {

        bannerDiv.classList.remove('hidden');

        bannerSpan.innerHTML = upcoming.map(apt => `${apt.title} (${new Date(apt.datetime).toLocaleDateString()})`).join(' · ');

      } else if (bannerSpan) {

        bannerDiv.classList.add('hidden');

      }

    }

    function renderTasks() {

      const container = document.getElementById('tasksListContainer');

      const emptyMsg = document.getElementById('emptyTasksMsg');

      if (!container) return;

      const tasks = [...appData.tasks];

      if (tasks.length === 0) {

        container.innerHTML = '';

        emptyMsg.classList.remove('hidden');

        return;

      }

      emptyMsg.classList.add('hidden');

      container.innerHTML = tasks.map(task => `

        <div class="flex items-center justify-between bg-gray-50 rounded-xl p-2.5 border border-gray-200">

          <div class="flex items-center gap-3 flex-1">

    function exportData() {

      const dataStr = JSON.stringify(appData, null, 2);

      const blob = new Blob([dataStr], {type: "application/json"});

      const url = URL.createObjectURL(blob);

      const a = document.createElement('a');

      a.href = url;

      a.download = `theline_team_data_${new Date().toISOString().slice(0,19)}.json`;

      document.body.appendChild(a);

      a.click();

      document.body.removeChild(a);

      URL.revokeObjectURL(url);

    }

    function importData(file) {

      const reader = new FileReader();

      reader.onload = (ev) => {

        try {

          const imported = JSON.parse(ev.target.result);

          if (imported.maintenanceLogs && imported.appointments && imported.tasks && imported.vehicles) {

            appData = imported;

            persistToLocalStorage();

            renderAll();

            alert("✅ Data imported successfully! Team members can now see updated fleet info.");

          } else { alert("Invalid file structure"); }

        } catch(e) { alert("Error parsing JSON"); }

      };

      reader.readAsText(file);

    }

    document.getElementById('exportDataBtn')?.addEventListener('click', exportData);

    document.getElementById('importFileInput')?.addEventListener('change', (e) => {

      if (e.target.files.length) importData(e.target.files[0]);

      e.target.value = '';

    });
